    // ===== 工具函数 =====

    /**
     * 安全的整数解析
     */
    function safeParseInt(value, defaultValue = 0) {
        const num = parseInt(value, 10);
        return isNaN(num) ? defaultValue : num;
    }

    /**
     * 获取指定范围内的随机整数 [min, max]
     */
    function getRandomInt(min, max) {
        return Math.floor(Math.random() * (max - min + 1)) + min;
    }

    /**
     * 从数组中随机选择一个元素
     */
    function getRandomElement(array) {
        if (!array || array.length === 0) return null;
        return array[Math.floor(Math.random() * array.length)];
    }

    /**
     * 格式化日期为 yyyy-mm-dd
     */
    function formatDate(date) {
        const year = date.getFullYear();
        const month = String(date.getMonth() + 1).padStart(2, '0');
        const day = String(date.getDate()).padStart(2, '0');
        return `${year}-${month}-${day}`;
    }

    /**
     * 解析日期字符串 yyyy-mm-dd
     */
    function parseDate(dateString) {
        try {
            const parts = dateString.split('-');
            if (parts.length !== 3) return null;
            return new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
        } catch (e) {
            console.error('[事件系统] 日期解析失败:', dateString, e);
            return null;
        }
    }

    /**
     * 获取当前日期（从变量中读取，如果不存在则使用系统日期）
     */
    function getCurrentDate(variables) {
        try {
            const currentDateStr = variables.world_info?.time?.current_date;
            if (currentDateStr) {
                const parsed = parseDate(currentDateStr);
                if (parsed) return parsed;
            }
        } catch (e) {
            console.warn('[事件系统] 无法从变量读取当前日期，使用系统日期:', e);
        }
        return new Date();
    }

    /**
     * 计算日期差（天数）
     */
    function daysBetween(date1, date2) {
        const oneDay = 24 * 60 * 60 * 1000;
        return Math.round((date2 - date1) / oneDay);
    }

    /**
     * 生成唯一的事件ID
     */
    function generateEventId() {
        return `event_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    }

    // ==================== MVU系统状态管理（对齐 外置状态栏.js） ====================
    const mvuState = {
        ready: false,
        mvu: null,
        check: () => {
            const win = window.parent || window;

            if (typeof win.Mvu === 'undefined') {
                mvuState.ready = false;
                mvuState.mvu = null;
                return false;
            }

            if (typeof win.Mvu.getMvuData !== 'function' ||
                typeof win.Mvu.replaceMvuData !== 'function') {
                mvuState.ready = false;
                mvuState.mvu = null;
                return false;
            }

            mvuState.ready = true;
            mvuState.mvu = win.Mvu;
            return true;
        }
    };

    let mvuWriteQueue = Promise.resolve();
    let mvuQueueLen = 0;
    const enqueueMvuWrite = (writeTask) => {
        mvuQueueLen++;
        const task = mvuWriteQueue.then(() => writeTask(), () => writeTask());
        mvuWriteQueue = task.catch(() => undefined);
        task.finally(() => {
            mvuQueueLen--;
            if (mvuQueueLen <= 0) {
                mvuQueueLen = 0;
                mvuWriteQueue = Promise.resolve();
            }
        });
        return task;
    };

    /**
     * 从 MVU 数据对象中提取实际游戏数据
     */
    function extractMvuGameData(mvuData) {
        if (mvuData?.stat_data && typeof mvuData.stat_data === 'object') {
            return mvuData.stat_data;
        }
        return {};
    }

    /**
     * 获取最新 MVU 游戏数据
     */
    function fetchLatestMvuData() {
        try {
            if (mvuState.ready || mvuState.check()) {
                const mvuData = mvuState.mvu.getMvuData({ type: 'message', message_id: 'latest' });
                if (mvuData?.stat_data && Object.keys(mvuData.stat_data).length > 0) {
                    return mvuData.stat_data;
                }
            }
        } catch (e) {
            console.error('[事件系统] 获取MVU数据失败:', e);
        }
        return {};
    }

    /**
     * 获取原始 MVU 数据容器（用于写回）
     */
    function fetchRawMvuData() {
        try {
            if (mvuState.ready || mvuState.check()) {
                let mvuData = mvuState.mvu.getMvuData({ type: 'message', message_id: 'latest' });
                if (!mvuData) {
                    mvuData = { stat_data: {} };
                }
                if (!mvuData.stat_data) {
                    mvuData.stat_data = {};
                }
                return mvuData;
            }
        } catch (e) {
            console.error('[事件系统] 获取原始MVU数据失败:', e);
        }
        return { stat_data: {} };
    }

    /**
     * 将游戏数据写回原始 MVU 容器
     */
    function applyGameDataToMvuContainer(mvuData, gameData) {
        if (!mvuData || typeof mvuData !== 'object') {
            return { stat_data: gameData };
        }
        if (!mvuData.stat_data || typeof mvuData.stat_data !== 'object') {
            mvuData.stat_data = {};
        }
        mvuData.stat_data = gameData;
        return mvuData;
    }

    /**
     * 持久化 MVU 数据
     */
    async function persistMvuGameData(mvuData) {
        try {
            const ok = await enqueueMvuWrite(async () => {
                if (!mvuState.ready && !mvuState.check()) {
                    console.error('[事件系统] MVU系统不可用');
                    return false;
                }

                await mvuState.mvu.replaceMvuData(mvuData, { type: 'message', message_id: 'latest' });
                const persistedData = extractMvuGameData(mvuData);
                console.log('[事件系统] 已写回MVU(message/latest)，当前事件数:', Object.keys(getCurrentEvents(persistedData)).length);
                return true;
            });
            return ok;
        } catch (e) {
            console.error('[事件系统] 持久化MVU数据失败:', e);
            return false;
        }
    }

    // ===== 事件生成逻辑 =====

    /**
     * 随机生成一个新事件
     */
    function generateRandomEvent(currentDate, availableTypes = EVENT_TYPES) {
        try {
            if (!availableTypes || availableTypes.length === 0) {
                console.log('[事件系统] 无可用事件类型，跳过生成');
                return null;
            }

            // 随机选择事件类型
            const eventType = getRandomElement(availableTypes);

            // 从对应类型的事件库中随机选择
            const eventPool = EVENT_DATABASE[eventType];
            const eventTemplate = getRandomElement(eventPool);

            if (!eventTemplate) {
                console.error('[事件系统] 事件模板为空，类型:', eventType);
                return null;
            }

            // 生成事件持续时间
            const duration = getRandomInt(CONFIG.MIN_EVENT_DURATION, CONFIG.MAX_EVENT_DURATION);

            // 计算开始和结束日期
            const startDate = new Date(currentDate);
            const endDate = new Date(currentDate);
            endDate.setDate(endDate.getDate() + duration);

            // 构造事件对象
            const event = {
                name: eventTemplate.name,
                type: eventType,
                description: eventTemplate.description,
                start_date: formatDate(startDate),
                remaining_duration: `${duration}天`,
                end_date: formatDate(endDate),
            };

            console.log('[事件系统] 成功生成事件:', {
                name: event.name,
                type: event.type,
                duration: duration,
                start: event.start_date,
                end: event.end_date,
            });

            return event;
        } catch (e) {
            console.error('[事件系统] 生成事件时发生异常:', e);
            return null;
        }
    }

    /**
     * 检查是否需要生成新事件
     */
    function shouldGenerateEvent(currentEventCount) {
        if (currentEventCount >= CONFIG.MAX_CONCURRENT_EVENTS) {
            console.log('[事件系统] 当前事件数已达上限，跳过生成');
            return false;
        }

        const roll = Math.random();
        const shouldGenerate = roll < CONFIG.EVENT_GENERATION_CHANCE;
        console.log(`[事件系统] 事件生成检定: ${roll.toFixed(3)} ${shouldGenerate ? '<' : '>='} ${CONFIG.EVENT_GENERATION_CHANCE} - ${shouldGenerate ? '通过' : '未通过'}`);
        return shouldGenerate;
    }

    // ===== 事件管理逻辑 =====

    /**
     * 获取当前所有事件
     */
    function getCurrentEvents(variables) {
        try {
            const events = variables.world_info?.events;
            if (!events || typeof events !== 'object') {
                return {};
            }

            // 过滤掉 $meta 和 $template 等系统字段
            const eventList = {};
            for (const key in events) {
                if (!key.startsWith('$') && Object.prototype.hasOwnProperty.call(events, key)) {
                    eventList[key] = events[key];
                }
            }

            return eventList;
        } catch (e) {
            console.error('[事件系统] 获取当前事件失败:', e);
            return {};
        }
    }

    /**
     * 检查并清理过期事件
     */
    function cleanupExpiredEvents(variables, currentDate) {
        const messages = [];
        let changed = false;

        try {
            const events = getCurrentEvents(variables);
            const eventsToDelete = [];

            for (const [eventId, event] of Object.entries(events)) {
                if (!event || !event.end_date) continue;

                const endDate = parseDate(event.end_date);
                if (!endDate) {
                    console.warn(`[事件系统] 事件 ${eventId} 的结束日期格式错误:`, event.end_date);
                    continue;
                }

                // 如果当前日期已经超过结束日期，标记为需要删除
                if (currentDate >= endDate) {
                    eventsToDelete.push({ id: eventId, name: event.name });
                }
            }

            // 执行删除操作
            if (eventsToDelete.length > 0) {
                eventsToDelete.forEach(item => {
                    delete variables.world_info.events[item.id];
                    messages.push(`📅 事件【${item.name}】已结束`);
                    console.log(`[事件系统] 删除过期事件: ${item.id} (${item.name})`);
                });
                changed = true;
            }
        } catch (e) {
            console.error('[事件系统] 清理过期事件时发生异常:', e);
        }

        return { messages, changed };
    }

    /**
     * 尝试生成并注册新事件
     */
    function tryGenerateNewEvent(variables, currentDate) {
        const messages = [];
        let changed = false;

        try {
            const currentEvents = getCurrentEvents(variables);
            const eventCount = Object.keys(currentEvents).length;

            console.log(`[事件系统] 当前活跃事件数: ${eventCount}/${CONFIG.MAX_CONCURRENT_EVENTS}`);

            // 检查是否应该生成新事件
            if (!shouldGenerateEvent(eventCount)) {
                return { messages, changed };
            }

            // 统计每种类型是否已存在
            const activeTypes = new Set(
                Object.values(currentEvents)
                    .filter(event => !!event)
                    .map(event => event.type)
            );

            const availableTypes = EVENT_TYPES.filter(type => !activeTypes.has(type));

            if (availableTypes.length === 0) {
                console.log('[事件系统] 所有事件类型均已在进行中，跳过生成');
                return { messages, changed };
            }

            // 生成事件
            const newEvent = generateRandomEvent(currentDate, availableTypes);
            if (!newEvent) {
                console.warn('[事件系统] 事件生成失败');
                return { messages, changed };
            }

            // 检查是否存在同名事件（避免重复）
            const isDuplicate = Object.values(currentEvents).some(
                event => event && event.name === newEvent.name
            );

            if (isDuplicate) {
                console.log(`[事件系统] 事件【${newEvent.name}】已存在，跳过生成`);
                return { messages, changed };
            }

            // 插入新事件
            const eventId = generateEventId();
            variables.world_info.events[eventId] = newEvent;
            messages.push(`🎊 新事件开始：【${newEvent.name}】- ${newEvent.description}`);
            console.log(`[事件系统] ✓ 成功注册新事件: ${eventId} (${newEvent.name})`);
            changed = true;
        } catch (e) {
            console.error('[事件系统] 生成新事件时发生异常:', e);
        }

        return { messages, changed };
    }

    // ===== 主处理函数 =====

    let updateCounter = 0;       // 更新计数器
    let isHandlingEvent = false; // 🛡️ 重入保护标志
    let lastDataSignature = '';
    let pollTimer = null;

    async function handleEventGeneration() {
        // 🛡️ 防止无限循环：如果正在处理事件，跳过
        if (isHandlingEvent) {
            console.warn('[事件系统] ⚠️ 检测到重入调用，跳过以防止无限循环');
            return;
        }

        isHandlingEvent = true;

        try {
            const mvuData = fetchRawMvuData();
            const variables = extractMvuGameData(mvuData);

            if (!variables || Object.keys(variables).length === 0) {
                console.warn('[事件系统] 无法获取MVU变量状态，跳过事件处理');
                return;
            }

            const currentSignature = JSON.stringify(variables);
            if (currentSignature === lastDataSignature) {
                return;
            }

            // 增加计数器
            updateCounter++;

            console.log(`[事件系统] 检测到MVU数据变化 (计数: ${updateCounter})`);
            lastDataSignature = currentSignature;

            // 检查是否需要执行（避免过于频繁）
            if (updateCounter % CONFIG.EVENT_CHECK_INTERVAL !== 0) {
                console.log(`[事件系统] 未达到检查间隔 (${updateCounter} % ${CONFIG.EVENT_CHECK_INTERVAL} = ${updateCounter % CONFIG.EVENT_CHECK_INTERVAL})，跳过本次`);
                return;
            }

            console.log(`[事件系统] ========== 第 ${updateCounter} 次更新 - 执行事件检查 ==========`);

            // 检查必要的变量结构是否存在
            if (!variables.world_info) {
                console.warn('[事件系统] world_info 不存在，跳过事件处理');
                console.warn('[事件系统] 当前读到的顶层键:', Object.keys(variables));
                console.warn('[事件系统] 提示: 请确保已在角色卡中使用 <VariableInsert> 初始化 world_info 结构');
                return;
            }

            // 检查events结构
            if (!variables.world_info.events) {
                console.warn('[事件系统] world_info.events 不存在，尝试初始化...');
                variables.world_info.events = {};
                applyGameDataToMvuContainer(mvuData, variables);
                await persistMvuGameData(mvuData);
                lastDataSignature = JSON.stringify(variables);
                console.log('[事件系统] 已完成 events 结构初始化');
                return;
            }

            // 获取当前日期
            const currentDate = getCurrentDate(variables);
            console.log('[事件系统] 当前日期:', formatDate(currentDate));

            // 1. 清理过期事件
            const cleanupResult = cleanupExpiredEvents(variables, currentDate);

            // 2. 尝试生成新事件
            const generationResult = tryGenerateNewEvent(variables, currentDate);

            const hasChanges = cleanupResult.changed || generationResult.changed;

            if (hasChanges) {
                applyGameDataToMvuContainer(mvuData, variables);
                await persistMvuGameData(mvuData);
                lastDataSignature = JSON.stringify(variables);
            }

            // 3. 输出消息（注意：不要尝试修改variables，因为这是只读的快照）
            const allMessages = [...cleanupResult.messages, ...generationResult.messages];
            if (allMessages.length > 0) {
                console.log(`[事件系统] 事件更新: ${allMessages.join(' | ')}`);
            }

            console.log('[事件系统] ========== 检查完成 ==========');
        } catch (e) {
            console.error('[事件系统] 主处理函数发生严重异常:', e);
            console.error('[事件系统] 错误堆栈:', e.stack);
        } finally {
            // 🛡️ 延迟重置标志，确保同一事件链中不会重复触发
            setTimeout(() => {
                isHandlingEvent = false;
            }, 200);
        }
    }

    // ===== 初始化函数 =====

    /**
     * 初始化事件系统
     */
    function initializeEventSystem() {
        console.log('[事件系统] 执行初始化...');

        // 设置计数器，使第一次检测到数据变化时就能触发检查
        updateCounter = CONFIG.EVENT_CHECK_INTERVAL - 1;
        console.log(`[事件系统] 计数器已初始化为 ${updateCounter}`);
        console.log(`[事件系统] 等待MVU数据变化触发检查（轮询间隔 ${CONFIG.POLL_INTERVAL_MS}ms）...`);

        if (pollTimer) {
            clearInterval(pollTimer);
        }
        pollTimer = setInterval(() => {
            handleEventGeneration().catch(e => {
                console.error('[事件系统] 轮询处理失败:', e);
            });
        }, CONFIG.POLL_INTERVAL_MS);

        handleEventGeneration().catch(e => {
            console.error('[事件系统] 初始检查失败:', e);
        });
    }

    // ===== 注册事件监听器 =====

    try {
        console.log('[事件系统] ========================================');
        console.log('[事件系统] 碧蓝档案随机事件生成器 v1.1 已加载');
        console.log('[事件系统] 🛡️ 重入保护: 已启用');
        console.log('[事件系统] 配置信息:');
        console.log(`[事件系统]   - 数据来源: MVU`);
        console.log(`[事件系统]   - 检测方式: 轮询最新MVU数据`);
        console.log(`[事件系统]   - 最大同时事件数: ${CONFIG.MAX_CONCURRENT_EVENTS}`);
        console.log(`[事件系统]   - 检查间隔: 每 ${CONFIG.EVENT_CHECK_INTERVAL} 次更新`);
        console.log(`[事件系统]   - MVU轮询间隔: ${CONFIG.POLL_INTERVAL_MS}ms`);
        console.log(`[事件系统]   - 生成概率: ${(CONFIG.EVENT_GENERATION_CHANCE * 100).toFixed(0)}%`);
        console.log(`[事件系统]   - 事件持续时间: ${CONFIG.MIN_EVENT_DURATION}-${CONFIG.MAX_EVENT_DURATION} 天`);
        console.log(`[事件系统]   - 总力战事件池: ${EVENT_DATABASE['总力战'].length} 个`);
        console.log(`[事件系统]   - 特殊事件池: ${EVENT_DATABASE['特殊事件'].length} 个`);
        console.log(`[事件系统]   - 节日事件池: ${EVENT_DATABASE['节日'].length} 个`);
        console.log('[事件系统] ========================================');

        // 执行初始化
        initializeEventSystem();
    } catch (e) {
        console.error('[事件系统] 注册事件监听器失败:', e);
        console.error('[事件系统] 错误堆栈:', e.stack);
    }
})();
