    function initStudentsGrid() {
      const characters = Object.keys(characterScenarios);

    // 切换开局
    function switchScenario(direction) {
      if (!selectedCharacter) return;

      const scenarios = characterScenarios[selectedCharacter];
      if (!Array.isArray(scenarios) || scenarios.length <= 1) return;

      if (direction === 'prev') {
        currentScenarioIndex = (currentScenarioIndex - 1 + scenarios.length) % scenarios.length;
      } else if (direction === 'next') {
        currentScenarioIndex = (currentScenarioIndex + 1) % scenarios.length;
      }

      updateScenarioSwitcher();

      // 更新立绘（如果有多开局封面）
      updateCharacterArtwork(selectedCharacter, currentScenarioIndex);

      // 生成提示消息

      let toastMessage;

        toastMessage = `已切换到${customLabels[currentScenarioIndex].main}`;
      } else {
        toastMessage = `已切换到开局${['一', '二', '三', '四', '五'][currentScenarioIndex] || (currentScenarioIndex + 1)}`;
      }
      showToast(toastMessage, 'success');
