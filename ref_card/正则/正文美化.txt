    // 安全解码URL函数
    function safeDecodeURIComponent(str) {
      try {
        return decodeURIComponent(str);
      } catch (e) {
        // 如果解码失败，尝试用另一种方式或返回原字符串
        try {
          return decodeURI(str);
        } catch (e2) {
          return str;
        }
      }
    }

    window.refreshImage = function (imageId, event) {
      const imageContent = document.getElementById(imageId);
      const img = imageContent.querySelector('img');
      if (!img) return;

      // 获取刷新按钮并添加旋转动画
      const button = event ? event.target.closest('.image-refresh-btn') : null;
      if (button) {
        button.style.transform = 'rotate(360deg)';
        button.style.transition = 'transform 0.5s ease';
        setTimeout(() => {
          button.style.transform = '';
        }, 500);
      }

      // 添加加载状态视觉反馈
      img.style.opacity = '0.5';
      img.style.transition = 'opacity 0.3s ease';

      const currentSrc = img.src;

      try {
        // 按/分割URL路径，找到NSFW/SFW之后的部分
        const parts = currentSrc.split('/');
        const nsfwIndex = parts.findIndex(p => p === 'NSFW' || p === 'SFW');

        if (nsfwIndex === -1 || nsfwIndex + 2 >= parts.length) {
          console.log('URL格式不正确:', currentSrc);
          return;
        }

        const category = parts[nsfwIndex]; // NSFW 或 SFW
        const characterEncoded = parts[nsfwIndex + 1]; // 角色名（编码）
        const fileNameEncoded = parts[nsfwIndex + 2]; // 文件名（编码），如 %E9%BB%98%E8%AE%A46.webp

        // 对每个路径段分别解码
        const character = safeDecodeURIComponent(characterEncoded);
        const fileNameWithExt = safeDecodeURIComponent(fileNameEncoded); // 如：默认6.webp

        // 去掉 .webp 扩展名
        const fileName = fileNameWithExt.replace(/\.webp$/, ''); // 如：默认6

        // 从文件名中分离类型和编号
        const fileMatch = fileName.match(/^(.+?)(\d+)$/);
        if (!fileMatch) {
          console.log('文件名格式错误:', fileName);
          return;
        }

        const type = fileMatch[1]; // 如：默认
        const currentNumber = fileMatch[2]; // 如：6

        // 从图片列表获取最大编号
        const categoryData = IMAGE_LISTS[category];
        if (!categoryData) {
          console.log('未找到分类:', category);
          return;
        }

        // 尝试精确匹配角色名，如果失败则尝试规范化匹配
        let characterData = categoryData[character];
        if (!characterData) {
          // 规范化字符串（去除空格、统一编码）
          const normalizedChar = character.trim().normalize('NFC');
          characterData = categoryData[normalizedChar];

          if (!characterData) {
            // 尝试匹配所有角色名（忽略空格）
            const availableChars = Object.keys(categoryData);
            const matched = availableChars.find(c =>
              c.trim().normalize('NFC') === normalizedChar ||
              c.replace(/\s/g, '') === character.replace(/\s/g, '')
            );

            if (matched) {
              characterData = categoryData[matched];
            } else {
              return; // 未找到角色，静默失败
            }
          }
        }

        // 尝试精确匹配图片类型，如果失败则尝试规范化匹配
        let maxNumber = characterData[type];
        if (!maxNumber) {
          const normalizedType = type.trim().normalize('NFC');
          maxNumber = characterData[normalizedType];

          if (!maxNumber) {
            // 尝试匹配所有类型名
            const availableTypes = Object.keys(characterData);
            const matched = availableTypes.find(t =>
              t.trim().normalize('NFC') === normalizedType ||
              t.replace(/\s/g, '') === type.replace(/\s/g, '')
            );

            if (matched) {
              maxNumber = characterData[matched];
            } else {
              return; // 未找到类型，静默失败
            }
          }
        }

        // 生成随机编号（1到maxNumber之间），确保不与当前编号相同
        let newNumber;
        if (maxNumber === 1) {
          // 如果只有1张图，则保持不变
          newNumber = 1;
        } else {
          // 生成不同于当前编号的随机数
          do {
            newNumber = Math.floor(Math.random() * maxNumber) + 1;
          } while (newNumber === parseInt(currentNumber));
        }

        // 重新构建URL，保持URL编码格式
        // 将新的类型+编号组合进行编码
        const newFileName = type + newNumber + '.webp';
        const newFileNameEncoded = encodeURIComponent(newFileName);

        // 重新构建完整URL（替换最后一个路径段）
        parts[nsfwIndex + 2] = newFileNameEncoded;
        const newSrc = parts.join('/');

        // 更新图片，并在加载完成后恢复透明度
        img.onload = function () {
          img.style.opacity = '1';
          // 应用当前图片大小设置
          img.style.width = imageSize + '%';
          img.style.maxWidth = imageSize + '%';
        };
        img.onerror = function () {
          img.style.opacity = '1'; // 即使加载失败也恢复透明度
        };
        img.src = newSrc;
      } catch (e) {
        console.error('刷新图片失败:', e);
      }
    };

    const mvuState = {
      ready: false,
      mvu: null,
      check: () => {
        const win = window.parent || window;
        if (typeof win.Mvu === 'undefined') {
          mvuState.ready = false;
          mvuState.mvu = null;
          return false;
        }
        if (typeof win.Mvu.getMvuData !== 'function' || typeof win.Mvu.replaceMvuData !== 'function') {
          mvuState.ready = false;

          return false;
        }
        mvuState.ready = true;
        mvuState.mvu = win.Mvu;
        return true;
      }

    function fetchLatestMvuData() {
      try {
        if (mvuState.ready || mvuState.check()) {
          const mvuData = mvuState.mvu.getMvuData({ type: 'message', message_id: 'latest' });
          if (mvuData?.stat_data && Object.keys(mvuData.stat_data).length > 0) {
            return mvuData.stat_data;
          }
        }
      } catch (e) {
        console.warn('[碧蓝档案正文美化] MVU读取失败:', e);
      }
      return {};
    }

    function applyMvuPatches(patches) {
      try {
        if (!mvuState.ready && !mvuState.check()) return false;
        const mvuData = mvuState.mvu.getMvuData({ type: 'message', message_id: 'latest' });
        if (!mvuData?.stat_data) return false;
        patches.forEach(patch => {
          const parts = String(patch.path || '').split('/').filter(Boolean);
          if (patch.op === 'replace' || patch.op === 'add') {
            let obj = mvuData.stat_data;
            for (let i = 0; i < parts.length - 1; i++) {
              if (!obj[parts[i]]) obj[parts[i]] = {};
              obj = obj[parts[i]];
            }
            obj[parts[parts.length - 1]] = patch.value;
          } else if (patch.op === 'remove') {
            let obj = mvuData.stat_data;
            for (let i = 0; i < parts.length - 1; i++) {
              if (!obj[parts[i]]) return;
              obj = obj[parts[i]];
            }
            delete obj[parts[parts.length - 1]];
          }
        });
        mvuState.mvu.replaceMvuData(mvuData, { type: 'message', message_id: 'latest' });
        return true;
      } catch (e) {
        console.warn('[碧蓝档案正文美化] MVU patch失败:', e);
        return false;
      }
    }

    function hashNameToColor(name) {
      let hash = 0;
      const input = String(name || '');
      for (let i = 0; i < input.length; i++) {
        hash = ((hash << 5) - hash) + input.charCodeAt(i);
        hash |= 0;
      }
      return DIALOGUE_COLOR_PALETTE[Math.abs(hash) % DIALOGUE_COLOR_PALETTE.length];
    }

    function lightenColor(hex, amount) {
      const num = parseInt(String(hex || '').replace('#', ''), 16);
      if (!Number.isFinite(num)) return hex;
      let r = Math.min(255, (num >> 16) + amount);
      let g = Math.min(255, ((num >> 8) & 0xff) + amount);
      let b = Math.min(255, (num & 0xff) + amount);
      return '#' + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);
    }

    const AVATAR_UPLOAD_MAX_BYTES = 5 * 1024 * 1024;
    const AVATAR_STORAGE_TARGET_BYTES = 900 * 1024;
    const AVATAR_MAX_DIMENSION = 1024;
    const AVATAR_MIN_DIMENSION = 256;

    function estimateDataUrlBytes(dataUrl) {
      const commaIndex = dataUrl.indexOf(',');
      if (commaIndex < 0) return 0;
      const base64 = dataUrl.slice(commaIndex + 1);
      const padding = (base64.match(/=+$/) || [''])[0].length;
      return Math.floor(base64.length * 3 / 4) - padding;
    }

    function encodeAvatarDataUrl(img) {
      const srcMaxSide = Math.max(img.width, img.height);
      let scale = srcMaxSide > AVATAR_MAX_DIMENSION ? AVATAR_MAX_DIMENSION / srcMaxSide : 1;
      let fallbackDataUrl = '';
      while (scale > 0) {
        const w = Math.max(1, Math.round(img.width * scale));
        const h = Math.max(1, Math.round(img.height * scale));
        const canvas = document.createElement('canvas');
        canvas.width = w;
        canvas.height = h;
        const ctx = canvas.getContext('2d');
        if (!ctx) break;
        ctx.imageSmoothingEnabled = true;
        ctx.imageSmoothingQuality = 'high';
        ctx.drawImage(img, 0, 0, w, h);
        const qualitySteps = [0.98, 0.95, 0.92, 0.88, 0.84, 0.8];
        let bestAtThisScale = '';
        for (const q of qualitySteps) {
          const candidate = canvas.toDataURL('image/webp', q);
          bestAtThisScale = candidate;
          if (estimateDataUrlBytes(candidate) <= AVATAR_STORAGE_TARGET_BYTES) {
            return candidate;
          }
        }
        fallbackDataUrl = bestAtThisScale;
        if (Math.max(w, h) <= AVATAR_MIN_DIMENSION) break;
        scale *= 0.85;
      }
      return fallbackDataUrl;
    }

    function initAvatarUpload() {
      const uploadBtn = document.getElementById('customCharUploadBtn');
      const fileInput = document.getElementById('customCharFileInput');
      const preview = document.getElementById('customCharAvatarPreview');
      if (!uploadBtn || !fileInput) return;
      uploadBtn.addEventListener('click', () => fileInput.click());
      fileInput.addEventListener('change', function () {
        const file = this.files && this.files[0];
        if (!file) return;
        if (!file.type.startsWith('image/')) { showSettingsMessage('请选择图片文件'); this.value = ''; return; }
        if (file.size > AVATAR_UPLOAD_MAX_BYTES) { showSettingsMessage('图片不能超过5MB'); this.value = ''; return; }
        const reader = new FileReader();
        reader.onload = function (e) {
          const img = new Image();
          img.onload = function () {
            const originalDataUrl = String(e.target?.result || '');
            const needTranscode = file.size > AVATAR_STORAGE_TARGET_BYTES || Math.max(img.width, img.height) > AVATAR_MAX_DIMENSION;
            _pendingAvatarData = needTranscode ? encodeAvatarDataUrl(img) : originalDataUrl;
            if (!_pendingAvatarData) {
              showSettingsMessage('头像处理失败，请换一张图片');
              fileInput.value = '';
              return;
            }
            preview.src = _pendingAvatarData;
            preview.style.display = 'block';
            uploadBtn.textContent = '✅ 已选择';
          };
          img.onerror = function () {
            showSettingsMessage('无法读取图片，请换一张');
            fileInput.value = '';
          };
          img.src = String(e.target?.result || '');
        };
        reader.readAsDataURL(file);
      });
    }

    function normalizeDisplayName(name) {
      const raw = String(name || '').trim();
      if (!raw) return '';
      const exact = Object.keys(CHARACTER_CONFIG).find(key => key === raw);
      if (exact) return exact;
      const aliasMatch = Object.entries(CHARACTER_CONFIG).find(([, config]) =>
        Array.isArray(config.aliases) && config.aliases.includes(raw)
      );
      return aliasMatch ? aliasMatch[0] : raw;
    }

    }

    function decodeHtmlEntities(text) {
      const textarea = document.createElement('textarea');

    function generateStyledNameHtml(config, normalizedName) {
      if (!config) {
        return `<span class="name-main">${normalizedName}</span>`;
      }

      const initialClass = `name-initial name-initial-${normalizedName}`;
      const isEnlarge = config.enlargeInitial === true;

      if (config.family_name) {
        const fName = config.family_name;
        const gName = config.given_name;
        if (isEnlarge) {
          let html = `<span class="${initialClass} name-initial-enlarge">${fName[0]}</span>`;
          if (fName.length > 1) html += `<span class="name-normal">${fName.slice(1)}</span>`;
          html += `<span class="name-initial-enlarge" style="font-weight:900">${gName[0]}</span>`;
          if (gName.length > 1) html += `<span class="name-normal">${gName.slice(1)}</span>`;
          return html;
        } else {
          let html = `<span class="${initialClass}">${fName[0]}</span>`;
          if (fName.length > 1) html += `<span class="name-normal">${fName.slice(1)}</span>`;
          html += `<span class="name-normal">${gName}</span>`;
          return html;
        }
      } else if (config.given_name) {
        const gName = config.given_name;
        let html = `<span class="${initialClass}">${gName[0]}</span>`;
        if (gName.length > 1) html += `<span class="name-normal">${gName.slice(1)}</span>`;
        return html;
      } else {
        if (normalizedName.length > 0) {
          return `<span class="${initialClass}">${normalizedName[0]}</span><span class="name-normal">${normalizedName.slice(1)}</span>`;
        }
        return `<span class="name-main">${normalizedName}</span>`;
      }
    }

    function processDialogueContent(dialogue) {
      dialogue = String(dialogue || '').replace(/\{[^}]+\}/g, '').trim();
      const patterns = [
        /「([^」]+)」（([^）]+)）/g,
        /「([^」]+)」\(([^)]+)\)/g,
        /『([^』]+)』（([^）]+)）/g,
        /【([^】]+)】（([^）]+)）/g
      ];
      let result = dialogue;
      patterns.forEach((regex) => {
        result = result.replace(regex, (match, text1, text2) => {
          return `<div class="bilingual-dialogue"><div class="primary-text">「${escapeHtmlText(text1)}」</div><div class="dialogue-divider"></div><div class="secondary-text">（${escapeHtmlText(text2)}）</div></div>`;
        });
      });
      if (result === dialogue && !/[「『【“"]/.test(dialogue)) {
        result = `「${escapeHtmlText(dialogue)}」`;
      }
      return result;
    }

    function wrapInlineDialogueQuote(segment) {
      if (!segment) return '';
      const pairs = { '「': '」', '『': '』', '【': '】', '“': '”', '"': '"' };
      const open = segment[0];
      const close = segment[segment.length - 1];
      if (pairs[open] !== close) return escapeHtmlText(segment);
      const inner = segment.slice(1, -1);
      return `<span class="inline-dialogue-quote"><span class="quote-mark">${escapeHtmlText(open)}</span>${escapeHtmlText(inner)}<span class="quote-mark">${escapeHtmlText(close)}</span></span>`;
    }

    function highlightInlineDialogueQuotes(text) {
      const source = decodeHtmlEntities(String(text || ''));
      const quoteRegex = /「[^」\n]+」|『[^』\n]+』|【[^】\n]+】|“[^”\n]+”|"[^"\n]+"/g;
      let result = '';
      let lastIndex = 0;
      let match;
      while ((match = quoteRegex.exec(source)) !== null) {
        result += escapeHtmlText(source.slice(lastIndex, match.index));
        result += wrapInlineDialogueQuote(match[0]);
        lastIndex = match.index + match[0].length;
      }
      result += escapeHtmlText(source.slice(lastIndex));
      return result;
    }

    function createDialogueElement(name, dialogue) {

    }

    function processTextContent() {
      const contentDiv = document.getElementById('readingContent');
      if (!contentDiv) return;

      const initialHtml = contentDiv.innerHTML.trim();
      if (!initialHtml) return;

      const workDiv = document.createElement('div');
      workDiv.innerHTML = initialHtml;
      contentDiv.setAttribute('aria-busy', 'true');

      const preservedBlocks = [];
      let preservedBlockIndex = 0;
      const standaloneHtmlBlockSelector = [
        ':scope > div[style]',
        ':scope > div[class]',
        ':scope > div[data-preserve-html]',
        ':scope > section[style]',
        ':scope > section[class]',
        ':scope > section[data-preserve-html]',
        ':scope > article[style]',
        ':scope > article[class]',
        ':scope > article[data-preserve-html]',
        ':scope > blockquote[style]',
        ':scope > blockquote[class]',
        ':scope > blockquote[data-preserve-html]',
        ':scope > figure[style]',
        ':scope > figure[class]',
        ':scope > figure[data-preserve-html]',
        ':scope > table[style]',
        ':scope > table[class]',
        ':scope > table[data-preserve-html]',
        ':scope > button',
        ':scope > a',
        ':scope > input',
        ':scope > textarea',
        ':scope > select'
      ].join(', ');

      Array.from(workDiv.querySelectorAll(standaloneHtmlBlockSelector)).forEach(blockEl => {
        if (blockEl.parentNode !== workDiv) return;
        const ph = '\n___HTMLBLOCK_' + (preservedBlockIndex++) + '___\n';
        const hasImage = !!blockEl.querySelector('img');
        workDiv.insertBefore(document.createTextNode(ph), blockEl);
        workDiv.removeChild(blockEl);
        preservedBlocks.push(
          hasImage && !blockEl.classList.contains('image-container')
            ? { ph: ph.trim(), html: createInteractiveImageContainer(blockEl.outerHTML) }
            : { ph: ph.trim(), wrapper: blockEl }
        );
      });

      let rawText = workDiv.innerHTML.trim();
      if (!rawText) {
        contentDiv.removeAttribute('aria-busy');
        return;
      }

      const tagPatterns = [
        {
          regex: /<DiceEvent>([\s\S]*?)<\/DiceEvent>/gi,
          title: '判定日志',
          type: 'dice-log'
        },
        {
          regex: /<Experience[-_]?Log>([\s\S]*?)<\/Experience[-_]?Log>/gi,
          title: '经验日志',
          type: 'exp-log'
        },
        {
          regex: /<QuestContract>([\s\S]*?)<\/QuestContract>/gi,
          title: '任务日志',
          type: 'quest-log'
        },
        {
          regex: /<CombatLog>([\s\S]*?)<\/CombatLog>/gi,
          title: '战斗日志',
          type: 'combat-log'
        },
        {
          regex: /<LootLog>([\s\S]*?)<\/LootLog>/gi,
          title: '战利品日志',
          type: 'loot-log'
        },
        {
          regex: /<Image>([\s\S]*?)<\/Image>/gi,
          title: '',
          type: 'image-tag'
        }
      ];
      const matches = [];
      tagPatterns.forEach(config => {
        let match;
        config.regex.lastIndex = 0;
        while ((match = config.regex.exec(rawText)) !== null) {
          matches.push({
            index: match.index,
            length: match[0].length,
            content: match[1].trim(),
            title: config.title,
            type: config.type
          });
        }
      });
      matches.sort((a, b) => a.index - b.index);
      const htmlParts = [];
      let lastIndex = 0;
      matches.forEach(match => {
        const beforeText = rawText.substring(lastIndex, match.index);
        if (beforeText.trim()) {
          htmlParts.push(processParagraphs(beforeText));
        }
        if (match.type === 'image-tag') {
          htmlParts.push(createImageElement(match.content));
        } else {
          htmlParts.push(createLogCollapsible(match.content, match.title, match.type));
        }
        lastIndex = match.index + match.length;
      });
      const remainingText = rawText.substring(lastIndex);
      if (remainingText.trim()) {
        htmlParts.push(processParagraphs(remainingText));
      }
      workDiv.innerHTML = htmlParts.join('');

      if (preservedBlocks.length > 0) {
        const walker = document.createTreeWalker(workDiv, NodeFilter.SHOW_TEXT, null, false);
        const textNodes = [];
        let node;
        while ((node = walker.nextNode())) textNodes.push(node);

        for (const block of preservedBlocks) {
          for (const tn of textNodes) {
            if (!tn.textContent.includes(block.ph)) continue;
            const parent = tn.parentNode;
            const trimmedContent = tn.textContent.replace(block.ph, '').trim();
            const fragment = document.createDocumentFragment();
            if (block.html) {
              const temp = document.createElement('div');
              temp.innerHTML = block.html;
              while (temp.firstChild) fragment.appendChild(temp.firstChild);
            } else {
              fragment.appendChild(block.wrapper);
            }

            if (parent.tagName === 'P' && !trimmedContent) {
              parent.parentNode.insertBefore(fragment, parent);
              parent.parentNode.removeChild(parent);
            } else {
              const parts = tn.textContent.split(block.ph);
              if (parts[0]) parent.insertBefore(document.createTextNode(parts[0]), tn);
              parent.insertBefore(fragment, tn);
              if (parts[1]) parent.insertBefore(document.createTextNode(parts[1]), tn);
              parent.removeChild(tn);
            }
            break;
          }
        }
      }

      contentDiv.replaceChildren(...Array.from(workDiv.childNodes));
      contentDiv.removeAttribute('aria-busy');
    }
    function processParagraphs(text) {
      let cleanText = text.replace(/<\/?(?:DiceEvent|Experience[-_]?Log|QuestContract|Image)(?!\s)>/gi, '');
      const paragraphs = cleanText.split(/\n+/).filter(p => p.trim().length > 0);
      return paragraphs.map(p => processLineWithImages(p.trim())).join('');
    }
    function processLineWithImages(line) {
      const htmlLine = decodeHtmlEntities(line);
      const tempDiv = document.createElement('div');
      tempDiv.innerHTML = htmlLine;
      const decodedLine = tempDiv.textContent || tempDiv.innerText || htmlLine;

      const braceNamePattern = /^[\s]*\{([^}]+)\}[\s]*(.+)$/;
      const braceMatch = decodedLine.match(braceNamePattern);
      if (braceMatch) {
        const rawName = braceMatch[1].trim();
        const normalizedName = normalizeDisplayName(rawName);
        const config = CHARACTER_CONFIG[normalizedName];
        if (config) {
          return createDialogueElement(rawName, braceMatch[2].trim());
        }
        return `<p>${highlightInlineDialogueQuotes(braceMatch[2].trim())}</p>`;
      }

      const imgRegex = /<img[^>]*>/i;
      if (imgRegex.test(htmlLine)) {
        const div = document.createElement('div');
        div.innerHTML = htmlLine;

</div>
<div class="image-content${expandedClass}" id="${uniqueId}" style="text-align: center;">
${imgHtml}
</div>
</div>
`;
      } else {
        return `<p>${highlightInlineDialogueQuotes(decodedLine)}</p>`;
      }
